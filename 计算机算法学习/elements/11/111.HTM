void rbDeleteFixup(rbtree **root,rbtree *x)
{
    rbtree *w;
    while(x != *root && x->color == BLACK)
    {
        if(x == x->p->lchild)  //第4行
        {
            w = x->p->rchild;
            if(w->color == RED) //情况1
            {
                w->color = BLACK;
                x->p->color = RED;
                leftRotete(x->p,root);
                w = x->p->rchild;
            }
            if(w->color == BLACK && w->lchild->color == BLACK && w->rchild->color == BLACK)
            {//情况2
                w->color = RED;
                x = x->p;
            }
            else if (w->color == BLACK && w->lchild->color == RED && w->rchild->color == BLACK)
            {//情况3
                w->color = RED;
                rightRotete(w,root);
                w = x->p->rchild;
            }
            else if (w -> color == BLACK && w->rchild->color == RED)
            {//情况4
                w->color = x->p->color;
                x->p->color = BLACK;
                w->rchild->color = BLACK;
                leftRotete(x,root);
                x = *root;
            }
        }
        else
        {
            //与上面对称
        }
    x->color = BLACK;  //倒数第2行
}
